    # 封装wx.request()请求以及对API的管理

    以下两段代码是对微信小程序的request请求封装和API的管理，以此来提高代码的可维护性和可阅读性以及模块化文件等，同事也提升了开发的效率。

    # request.js

        /**
 * 封装微信的的request
 */

//  项目BASE_URL
const WxApiRoot = 'http://localhost:3000';

function request(url, data = {}, msg, method) {
  if (msg.length > 0) {
    wx.showLoading({
      title: msg // 数据请求前loading
    })
  }
  return new Promise(function (resolve, reject) {
    wx.request({
      url: WxApiRoot + url,
      data: data,
      method: method,
      timeout:8000,
      header: {
        'Content-Type': 'application/json',
        'token': wx.getStorageSync('token')
      },
      success: function (res) {
        setTimeout(()=>{
          if (msg.length > 0) {
          wx.hideLoading()
        }
        },1000)
        if (res.data.code == 401) {
          wxLogin()
        } else if (res.data.code !=200) {
          wx.showToast({
            title: res.data.msg,
            duration:3000,
            icon:'none'
          })
        }
        resolve(res.data)
      },
      fail: function (err) {
        if (err.errMsg=='request:fail timeout') {
          wx.showToast({
            title: '网络超时，请稍后再试',
            icon:'none',
            duration:3000
          })
        }
        wx.setStorageSync('token', '')
        if (msg.length > 0) {
          wx.hideLoading()
        }
        wx.showToast({
          title: err.msg,
          icon:'none',
          duration:3000
        })
        reject(err)
      }
    })
  });
}


function wxLogin(isNeed=true) {
  // 如果不想调用登录后执行 onload onShow 就传 false
  return new Promise(function (resolve, reject) {
    wx.login({
      success: function (res) {
        if (res.code) {
          var data = {
            code: res.code,
            storeCode: '9003'
          }
          post('ma/login/code2Session', data, '').then(res => {
            
            wx.setStorageSync('token', res.data.token);
            wx.setStorageSync('storeInfo',res.data)
            wx.setStorageSync('isMember',res.data.isMember)
            let pages = getCurrentPages();
            let currPage = null;
            if (pages.length) {
              currPage = pages[pages.length - 1];
            }
            if (isNeed) {
              currPage.onLoad(currPage.options)
              currPage.onShow()
            }
          
            resolve(true);
          })
        }
      },

    });
  });
}

function get(url, params, msg = '加载中...', isParams = "true") {
  return request(url, params, msg, 'GET')
}

function post(url, data, msg = '加载中...', isParams = "false") {
  return request(url, data, msg, 'POST')
}
module.exports = {
  get,
  post,
  wxLogin
}
